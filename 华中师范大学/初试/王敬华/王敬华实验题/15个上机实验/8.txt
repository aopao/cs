#include<stdio.h>

#include<string.h>
#define LEN sizeof(struct student)
struct student
{char Name[10];
float Maths;
float English;
float C;
float Total;
struct student*next;
};
int n;
struct student*creat()
{
	struct student*head,*p1,*p2;
	n=0;

	printf("student1:\nName\t");

	scanf("%f",&p1->C);
	p1->Total=(p1->Maths+p1->English+p1->C);
	head=NULL;
	while(p1->Total!=0)
	{
		n=n+1;
    if(n==1) head=p1;
		else
		{
			p2->next=p1;
			p2=p1;
			p1=(struct student*)malloc(LEN);
			printf("student%d:\nName\t",n);
			scanf("%s",p1->Name);
			printf("Maths\t");
			scanf("%f",&p1->Maths);
			printf("English\t");
			scanf("%f",&p1->English);
			printf(" C\t");
			scanf("%f",&p1->C);
			p1->Total=p1->Maths+p1->English+p1->C;
		}
	}
	p2->next=NULL;
	return(head);
}

			pt=pt->next;
		}while(pt!=NULL);
}
